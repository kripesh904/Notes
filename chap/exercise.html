1. C Program to check whether the input number is prime


        #include<stdio.h>
        
        int main() {
            int i, n, c = 0;
            printf("Enter any number : ");
            scanf("%d", &n);
            for(i = 2; i < n; i++) {
                if(n % i == 0) {
                    c++;
                }
            }
            if(c == 0) {
                printf("Prime");
            } else {
                printf("Composite");
            }
            return 0;
        }

2. C Program to find factorial of an input number


        #include<stdio.h>
        
        int main() {
            int i, n, a = 1;
            printf("Enter any number : ");
            scanf("%d", &n);
            for(i = 1; i <= n; i++) {
                a = a * i;
            }
            printf("Factorial = %d", a);
            return 0;
        }

3. C Program to check whether the input number is palindrome


        #include<stdio.h>
        
        int main() {
            int i, n, a, r = 0;
            printf("Enter any number : ");
            scanf("%d", &n);
            a = n;
            while(n != 0) {
                i = n % 10;
                r = r * 10 + i;
                n = n / 10;
            }
            if(a == r) {
                printf("Palindrome");
            } else {
                printf("Not a Palindrome");
            }
            return 0;
        }

Output

4. C Program to generate fibonacci Series


        #include<stdio.h>
        
        int main() {
            int i, n, a = 0, b = 1, c;
            printf("Enter any number : ");
            scanf("%d", &n);
            for(i = 1; i <= n; i++) {
                c = a + b;
                printf("%d ", a);
                a = b;
                b = c;
            }
            return 0;
        }
